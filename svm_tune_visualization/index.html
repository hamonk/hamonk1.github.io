# ROC Curves
library(ROCR)
rocplot = function(pred, truth, ...) {
  predob=prediction(pred,truth)
  perf=performance(predob,"tpr","fpr")
  plot(perf,...)
}

# we first plot the ROC curve with gamma=2 (that's the optimal value found earlier)
svmfit.opt=svm(y~., data=dat[train,], kernel ="radial", gamma=2, cost=1, decision.values=T)

# we asked for the decision values in the model so we have the parameters
# this will allow us to plot a graph, based on a threshold that decides to which class
# a value belongs
fitted=attributes(predict(svmfit.opt,dat[train,],decision.values=TRUE))$decision.values
par(mfrow=c(1,2))
rocplot(fitted,dat[train,"y"],main="Training Data")

# we plot the ROC curve for gamma=50 where there is overfitting
# the ROC curve is perfect....
svmfit.flex=svm(y~., data=dat[train ,], kernel ="radial",
                  gamma=50, cost=1, decision.values =T)
fitted=attributes (predict (svmfit.flex ,dat[train ,], decision.values=T))$decision.values
rocplot (fitted ,dat[train ,"y"],add=T,col="red ")

# now let's do the same on test data
fitted =attributes (predict (svmfit.opt ,dat[-train ,], decision.values=T))$decision.values
rocplot (fitted ,dat[-train ,"y"], main="Test Data")
# we now see that the gamma=50 algo is doing worse on the test data
fitted=attributes (predict (svmfit.flex ,dat[- train ,], decision.values=T))$decision.values
rocplot (fitted ,dat[-train ,"y"],add=T,col="red")
