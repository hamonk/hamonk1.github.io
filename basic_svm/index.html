# add features
x1_2 <- x1^2
x2_2 <- x2^2
X <- matrix(c(x1, x2, x1_2, x2_2), ncol=4)

# this is our prediction on the training set
plot(matrix(c(x1, x2), ncol=2), col=(3+y))
dat = data.frame(x=X, y=as.factor(y))

# run the basic svm algo
model <- svm(y ~., data=dat, kernel="polynomial", cost=5, scale=TRUE)

# predict our train set
predict_y <- as.numeric(as.character(predict(model, X)))

# plot prediction
points(x1, x2, main="Scatterplot Example", xlab="x1", ylab="x2", pch=3, col=(predict_y+1))

title(main="prediction is the cross, some errors")
